def tonum(s):
    l = len(s)
    num = ''
    for i in range(l//2):
        num += s[-2:]
        s = s[:-2]
    return int(num, 16)

def concode(num, a, b):
    global CF, ZF, SF, OF
    ZF = num == 0
    SF = num < 0
    OF = (a < 0 == b < 0) and (num < 0 != a < 0)
